❌ Buggy — Wrong Base Cases

// BUG 1: Both base cases return 0
function fibBuggy(n) {
  if (n <= 0) return 0;
  if (n === 1) return 0; // ← should be 1
  return fibBuggy(n-1) + fibBuggy(n-2);
}
// Produces: 0,0,0,0,0,0,0,0 …
// All arcs have radius 0 → invisible!

// BUG 2: Arc pivot never advances
let x = centerX, y = centerY;
arcs.forEach((r, i) => {
  ctx.arc(x, y, r, ...);
  // ← x, y never updated
  // Every arc drawn from same point
});

✅ Fixed — Correct Base Cases

// FIX 1: Proper Fibonacci base cases
function fibFixed(n) {
  if (n <= 0) return 0;
  if (n === 1) return 1; // ← correct
  return fibFixed(n-1) + fibFixed(n-2);
}
// Produces: 1,1,2,3,5,8,13,21 …

// FIX 2: Pivot advances each quarter-turn
arcs.forEach((r, i) => {
  ctx.arc(x, y, r, ...);
  // advance pivot by direction × radius
  x += dx(i) * r;
  y += dy(i) * r;
});
