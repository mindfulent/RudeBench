GET /users

Retrieve a list of all users.

Request:

GET /users HTTP/1.1

Response:

{
  "users": [
    {
      "id": 1,
      "name": "John Doe"
    },
    {
      "id": 2,
      "name": "Jane Doe"
    }
  ]
}

GET /users/{id}

Retrieve a user by ID.

Request:

GET /users/1 HTTP/1.1

Response:

{
  "id": 1,
  "name": "John Doe"
}

POST /users

Create a new user.

Request:

POST /users HTTP/1.1
Content-Type: application/json

{
  "name": "Jo"
}

Response:

{
  "id": 3,
  "name": "Jo"
}

PUT /users/{id}

Update an existing user.

Request:

PUT /users/1 HTTP/1.1
Content-Type: application/json

{
  "name": "John Smith"
}

Response:

{
  "id": 1,
  "name": "John Smith"
}

DELETE /users/{id}

Delete a user.

Request:

DELETE /users/5 HTTP/1.1

Response:

{
  "message": "User deleted successfully"
}

PATCH /users

Update multiple users.

Request:

PATCH /users HTTP/1.1
Content-Type: application/json

{
  "users": [
    {
      "id": 1,
      "name": "John Smith"
    },
    {
      "id": 2,
      "name": "Jane Smith"
    }
  ]
}

Response:

{
  "users": [
    {
      "id": 1,
      "name": "John Smith"
    },
    {
      "id": 2,
      "name": "Jane Smith"
    }
  ]
}
